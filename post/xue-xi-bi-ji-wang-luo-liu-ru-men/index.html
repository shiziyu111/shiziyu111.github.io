1.何为网络流

网络流是一种比较神奇的图，我们用你家楼下的水管来类比，其中有一个源点 sss（自来水厂），和汇点 ttt（你家），水管有容量 fif_ifi​，水流有流量。

（注：我画不来水管，请自行想象）

水流从自来水厂流向你家，注意：水流不会在经过某个水管之后就不占据容量了。

这就是最基础的网络流了。

2.最大流

那么问题来了，如何使自来水厂流向你家的单位水流最多？

Ford-Fulkerson 算法

有一种很显然的做法，即贪心，不断寻找从 sss 到 ttt 的路径即可，这样的路径也被我们称之为增广路。

但贪心存在一个问题，如我们刚才展示的图，假如我们的一条路径是 sss->111->222->ttt，那么我们 sss->222->ttt 的路径将会被堵塞（这下真的叫“以塞忠谏之路”了），而实际上 sss->111->ttt 和 sss->222->ttt 才是最优的路径。

因此我们需要有反悔的权利，如何反悔呢，很简单：建一条反向边，这样一来一回，约等于没有变化，于是我们就学会了 Ford-Fulkerson 算法，不断地从源点出发，寻找增广路，建立反向边即可。

但是 Ford-Fulkerson 算法慢的要死，还可以被卡，只需要我们在上图更改一下权值（没错，这个算法与权值有关），就可以卡掉，如下图（设 N=1145141919810N=1145141919810N=1145141919810）：

Ford-Fulkerson 算法会不断循环 sss->111->222->ttt 和 sss->222->111->ttt 的路径，这样你就炸了。

那看来我们需要寻找一个与权值没有太大关联的算法。

Edmonds-Karp 算法

Edmonds-Karp 算法相比于 FF 算法，使用了 BFS，而且找的是最短的（边数最少）增广路，我个人认为还是比较玄学的，应该是为了使遍历顺序较为有条理，比如这个图里会走 sss->111->ttt 和 sss->222->ttt，可以证明与边权无关（我不会证）。

Dinic 算法

Edmonds-Karp 算法虽然相对于 FF 算法较为优秀，但还是太慢了，每一次寻找增广路都要遍历一次图，就像你为了找到你丢掉的小说，在书堆里找一遍，又要为了找你妹丢掉的练习册，再在书堆里找一遍，为什么不统一找呢？于是就有了 Dinic 算法。

Dinic 的特色是在遇到某个点向其他边遍历时，如果还剩有流量，那么继续去寻找增广路（所以你就知道了它是 DFS）。但这样存在一个问题：假如你又出现了像 Ford-Fulkerson 算法一样乱序查找怎么办？很简单，先 BFS 一边以确定顺序呗（其实就是分个层），于是 Dinic 就同时有了 BFS 和 DFS。

于是 Dinic 能处理比较大规模的数据（一般是 10510^5105 以内），即使之后可能还要说 ISAP、HLPP 等等，Dinic 已经具备解决大部分题目的能力了（卡 Dinic 的出题人没浮木!），所以我们就先把后面的算法鸽一鸽 （其实是笔者也不会）。

Code

我只写了Dinic的代码。

#include<bits/stdc++.h>
using namespace std;
#define int long long
const int N=1e6+5,inf=1e9;
int n,m,s,t,ans;
int now[N],dis[N];//当前弧优化 
vector<int> a[N],g[N],oth[N];
//边，流，对边编号（相当于链式前向星的^1） 
void add(int u,int v,int w)
{
	a[u].push_back(v);
	g[u].push_back(w);
	a[v].push_back(u);
	g[v].push_back(0);
	oth[u].push_back(a[v].size()-1);
	oth[v].push_back(a[u].size()-1);
}
bool bfs()//判断是否合理并建分层图
{
	for(int i=1;i<=n*m;i++)
	{
		dis[i]=inf;
	}
	queue<int> q;
	q.push(s);
	dis[s]=0;
	now[s]=0;
	while(!q.empty())
	{
		int u=q.front();
		q.pop();
		for(int i=0;i<a[u].size();i++)
		{
			int v=a[u][i];
			if(g[u][i]>0&&dis[v]==inf)
			{
				dis[v]=dis[u]+1;
				now[v]=0;
				q.push(v);
				if(v==t)
				{
					return true;
				}
			}
		}
	}
	return false;
}
int dfs(int u,int value)//value指上一条路申请的额度 
{
	if(u==t) 
	{
		return value;
	}
	int res=0;//分走的流 
	for(int i=now[u];i<a[u].size()&&value!=0;i++)
	{
		now[u]=i;
		int v=a[u][i];
		int w=g[u][i];
		if(w>0&&dis[v]==dis[u]+1)
		{
			int k=dfs(v,min(value,w));
			if(k==0)
			{
				dis[v]=inf;
			}
			g[u][i]-=k;
			g[v][oth[u][i]]+=k;
			res+=k;
			value-=k;
		}
	}
	return res;
}
void dinic()
{
	while(bfs())
	{
		ans+=dfs(s,inf);
	}
}
void connect()
{
	for(int i=1;i<=m;i++)
	{
		int u,v,f;
		scanf("%lld%lld%lld",&u,&v,&f);
		add(u,v,f);
	}
}
signed main()
{
	scanf("%lld%lld%lld%lld",&n,&m,&s,&t);
	connect();
	dinic();
	printf("%lld",ans);
	return 0;
}

3.费用流

费用流相较于最大流，其实也就是多了一个单边费用 cic_ici​ ，表示每流过 111 单位的流量，就要为此支付 cic_ici​ 的费用，最小费用最大流其实也就是在保证流量最大的情况下，使费用尽可能的少（当然还有最大费用最大流，不过用的不多，而且在最小费用最大流的模板上改一下就好了）。

我们来看看如何求解。

其实我们首先为了迎合撤销的操作，所以也要把费用撤销掉，所以我们给所有相反边的费用取上原边的负值，然后我们如何知道费用最少的的流呢？

最短路算法！

没错，我们此时使用 SPFA 来代替此前的分层，跟着最短路走到汇点，就能较为贪心的算出最小费用（我们刚才建立了负边，所以不能用 dijsktra，但假如 SPFA 死了，也不是不能用，但很麻烦，具体请见 fengzhaoyu 大佬的文章，卡 SPFA 的出题人没浮木！）。

最后我们就能愉快的做出来了。

Code

#include<bits/stdc++.h>
using namespace std;
#define int long long
const int N=5e4+5,inf=1e18;
struct ed{
	int v;
	int f;//flow
	int c;//cost
};
int n,m,s,t,ans_w,ans_f;
vector<ed> a[N];
vector<int> oth[N];//对边编号
int now[N],dis[N],vis[N];
void add(int u,int v,int f,int c)
{
	a[u].push_back({v,f,c});
	a[v].push_back({u,0,-c});
	oth[u].push_back(a[v].size()-1);
	oth[v].push_back(a[u].size()-1);
}
bool spfa()
{
	for(int i=1;i<=n;i++)
	{
		dis[i]=inf;
		vis[i]=now[i]=0;
	}
	queue<int> q;
	q.push(s);
	vis[s]=1;
	dis[s]=0;
	while(!q.empty())
	{
		int u=q.front();
		q.pop();
		vis[u]=0;
		for(int i=0;i<(int)a[u].size();i++)
		{
			int v=a[u][i].v;
			int f=a[u][i].f;
			int w=a[u][i].c;
			if(f>0&&dis[v]>dis[u]+w)
			{
				dis[v]=dis[u]+w;
				if(!vis[v])
				{
					q.push(v);
					vis[v]=1;
				}
			}
		}
	}
	if(dis[t]==inf)
	{
		return false;
	}
	else
	{
		return true;
	}
}
int dfs(int u,int value)
{
	if(u==t)
	{
		return value;
	}
	vis[u]=1;
	int res=0;
	for(int i=now[u];i<(int)a[u].size()&&value!=0;i++)
	{
		now[u]=i;
		int v=a[u][i].v;
		int f=a[u][i].f;
		int w=a[u][i].c;
		if(vis[v]==0&&f>0&&dis[v]==dis[u]+w)
		{
			int flow=dfs(v,min(value,f));
			a[u][i].f-=flow;
			a[v][oth[u][i]].f+=flow;
			ans_w+=w*flow;
			res+=flow;
			value-=flow;
		}
	}
	vis[u]=0;
	return res;
}
void dinic()
{
	int res=0;
	while(spfa())
	{
			ans_f+=dfs(s,inf);
			
	}
}
void connect()
{
	for(int i=1;i<=m;i++)
	{
		int u,v,f,c;
		scanf("%lld%lld%lld%lld",&u,&v,&f,&c);
		add(u,v,f,c);
	}
}
signed main()
{
	scanf("%lld%lld%lld%lld",&n,&m,&s,&t);
	connect();
	dinic();
	printf("%lld %lld",ans_f,ans_w);
	return 0;
}

你也许会问我，我为什么要在这写这么一个看似无用的算法，但其实网络流非常的万用，就比如以下的网络流与线性规划24题，假如你打开这些题看一看，会发现它们基本上与网络流没什么关系，你可能会觉得它们是数论，是 DP，是图论，甚至其中有些题也存在更好想的非网络流做法，所以网络流的两大难点就是：

  1. 看出它是网络流。（这个反而是最难的）
  2. 清楚如何建图。

解决了这两个难点，你也许能轻松做出一道蓝题，紫题，甚至是黑，而且网络流也可以解决出大部分具有奇奇怪怪限制的省选题，比如下面：
